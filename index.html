Day 03

Lists & Loops

Efficiently managing collections and automating data filtering using Pythonic patterns.

Task Filtering (List Comprehension)
# Filtering high-priority tasks automatically
tasks = ["Email Client", "Fix Bug (High)", "Update Docs", "Deploy (High)"]
urgent = [t for t in tasks if "(High)" in t]
print(f"Urgent Actions: {urgent}")
Data Cleaning (Sets & Sorting)
# Removing duplicates and sorting values
raw_data = [45, 12, 89, 12, 45, 10]
cleaned = sorted(list(set(raw_data)))
print(f"Sorted Unique Data: {cleaned}")

Palindrome Checker
def is_palindrome(word):
    clean = word.replace(" ", "").lower()
    return clean == clean[::-1]
